--1 도서명과 출판사
SELECT BOOKNAME, PUBLISHER
FROM BOOK;
--2 전체 도서 정보
SELECT *
FROM BOOK
--3 만원 이상 도서명과 출판사
SELECT BOOKNAME, PUBLISHER
FROM BOOK
WHERE PRICE>=10000;
--4 가격이 만원 미만 도서명과 출판사
SELECT BOOKNAME, PUBLISHER
FROM BOOK
WHERE PRICE < 10000;
--5 가격이 만원이상 2만원 이하 도서정보
SELECT *
FROM BOOK
WHERE PRICE >=10000 AND PRICE<=20000;
--6 나무수 출판사의 도서정보
SELECT *
FROM BOOK
WHERE PUBLISHER ='나무수';
--7 도서번호 5번~10번까지 도서번호, 도서명, 가격
SELECT BOOKID, BOOKNAME, PRICE
FROM BOOK
WHERE BOOKID >=5 AND BOOKID <=10;

SELECT BOOKID, BOOKNAME, PRICE
FROM BOOK
WHERE BOOKID  BETWEEN 5 AND 10;
--8 도서명순으로 도서명과 가격 ASCENDING: 오름차순, DESCENDING: 내림차순
SELECT BOOKNAME, PRICE
FROM BOOK
ORDER BY BOOKNAME ASC;
--9 가격순으로 도서명과 가격
SELECT BOOKNAME, PRICE 
FROM BOOK
ORDER BY PRICE;
--10 가격역순으로 도서명과 가격
SELECT BOOKNAME, PRICE 
FROM BOOK
ORDER BY PRICE DESC;
--11 출판사가 나무수인 도서정보
SELECT *
FROM BOOK
WHERE PUBLISHER ='나무수';
--11-2 출판사가 나무수가 아닌 도서정보
SELECT *
FROM BOOK
WHERE NOT PUBLISHER = '나무수';
--12 출판사가 굿스포츠인 도서정보
SELECT *
FROM BOOK
WHERE PUBLISHER ='굿스포츠';
--13 출판사가 나무수, 굿스포츠인 도서정보
SELECT *
FROM BOOK
WHERE PUBLISHER ='나무수' OR PUBLISHER = '굿스포츠';
--13-2
SELECT *
FROM BOOK
WHERE PUBLISHER IN ('나무수' , '굿스포츠');
--14 주문정보
SELECT * 
FROM ORDERS;
--15 가격이 8000원 이하인 도서정보
SELECT *
FROM BOOK
WHERE PRICE <=8000;
--16 가격이 8000원 이하인 도서  수
SELECT COUNT(*)
FROM BOOK
WHERE PRICE <=8000;
--17 도서 가격이 8000원 이하이고 도서번호가 3이상인 도서정보
SELECT *
FROM BOOK
WHERE PRICE <=8000 AND BOOKID >=3;
--18 고객의 이름, 주소, 전화번호 
SELECT NAME, ADDRESS, PHONE
FROM CUSTOMER;
--19 도서정보와 주문정보
SELECT *
FROM BOOK, ORDERS
WHERE BOOK.BOOKID=ORDERS.BOOKID;
--20 고객정보와 주문정보
SELECT *
FROM CUSTOMER, ORDERS
WHERE CUSTOMER.CUSTID = ORDERS.CUSTID;
